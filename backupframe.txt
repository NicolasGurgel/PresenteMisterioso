document.getElementById("botao").onmouseover = function() {
    let largura = window.innerWidth;
    let altura = window.innerHeight;
    let x = Math.random() * (largura - this.offsetWidth);
    let y = Math.random() * (altura - this.offsetHeight);
    this.style.left = x + "px";
    this.style.top = y + "px";
}

const mario = document.getElementById("mario");
let dir = null;
let frame = null;
let iswalking = false;

function jump(){
    mario.classList.add("jump");
    setTimeout(function(){
        mario.classList.remove("jump");
    }, 500);
}

function walking(){
    if(dir==="r"){
        let atual = parseInt(window.getComputedStyle(mario).left, 10);
        mario.classList.remove("faceL");
        mario.classList.add("faceR");
        mario.style.left = (atual+50) + "px";
    }else if(dir==="l"){
        let atual = parseInt(window.getComputedStyle(mario).left, 10);
        mario.classList.remove("faceR");
        mario.classList.add("faceL");
        mario.style.left = (atual-50) + "px";
    }
    
    if(iswalking){
        frame = requestAnimationFrame(walking);
    }
}

function walk(direction){
    if(dir!==direction){
        dir=direction;
        if(!iswalking){
            iswalking=true;
            walking();
        }
    }
}

function stop(){
    iswalking = false;
    dir=null;
    if(frame){
        cancelAnimationFrame(frame);
        frame=null;
    }
}

document.addEventListener("keydown", (event)=>{
    if(event.code === 'Space'){
        jump();
    }else if(event.code === 'KeyD'){
        walk("r");
    }else if(event.code === 'KeyA'){
        walk("l");
    }
});

document.addEventListener("keyup", (event)=>{
    if(event.code==='KeyD'||event.code==='KeyA'){
        stop();
    }
});
